clear all;
close all;
clc;

n=-5:1:6;
x=[-3,2,5,6,0,3,5,9,-7,2,0,3];

stem(n,x);
xlim([-7 8]);
ylim([-12 12]);

xlabel('n');
ylabel('x(n)');
title('Stem Plot of Discrete Sequence')
